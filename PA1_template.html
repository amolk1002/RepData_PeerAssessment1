Now first Create a new dataset that is equal to the original dataset but with the missing data filled in.

newData <- data 
for (i in 1:nrow(newData)) {
    if (is.na(newData$steps[i])) {
        newData$steps[i] <- avgSteps[which(newData$interval[i] == avgSteps$interval), 
]$meanOfSteps
    }
}
